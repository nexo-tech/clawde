<pdf_generation>
```python
from reportlab.lib.pagesizes import letter
from reportlab.lib.styles import getSampleStyleSheet
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer, Image
import os

os.makedirs('files/reports', exist_ok=True)

doc = SimpleDocTemplate("files/reports/topic_report.pdf", pagesize=letter)
styles = getSampleStyleSheet()
story = []

# Add content...
story.append(Paragraph("Title", styles['Title']))
story.append(Paragraph("Content...", styles['Normal']))

# Add chart if exists
if os.path.exists('files/charts/chart.png'):
    story.append(Image('files/charts/chart.png', width=400, height=250))

doc.build(story)
```
